$$\begin{cases} \dfrac{dx}{dt} = -0.41x(t)- 0.89y(t)+sin(t+7)+1\\ \dfrac{dy}{dt} = -0.52x(t)- 0.61y(t)+cos(t+6)+1 \end{cases}$$

Модель боевых действий между регулярными войсками

function reg(u, p, t)
    x, y = u
    a, b, c, h = p
    dx = -a*x - b*y+sin(t+7)+1
    dy = -c*x -h*y+cos(t+6)+1
    return [dx, dy]
end

Модель ведение боевых действий с участием регулярных войск и партизанских отрядов

$$\begin{cases} \dfrac{dx}{dt} = -0.37x(t)-0.675y(t)+|2sin(t)|\\ \dfrac{dy}{dt} = -0.432x(t)y(t)-0.42y(t)+cos(t)+2 \end{cases}$$

Модель ведение боевых действий с участием регулярных войск и партизанских отрядов

function reg_part(u, p, t)
    x, y = u
    a, b, c, h = p
    dx = -a*x - b*y+abs(2 * sin(t))
    dy = -c*x*y -h*y+cos(t) + 2
    return [dx, dy]
end
